\begin{frame}
\frametitle{Symmetries of the carpet}
 \begin{itemize}[<+->]
\item Translation along the diagonal (generator $d$)
\item Even-step translation along the anti-diagonal (generator $\aa$)
\item Rotations over $180^\circ$ around each $(\exists,\stsixe)$-midpoint
\end{itemize}
\vspace*{1cm}
Do we really need infinitely many generators?              
\end{frame}


\begin{frame}%[fragile]
\frametitle{One generator suffices, group presentation}
\[
\begin{tikzcd}[ampersand replacement=\&, column sep={1cm, between origins}]
\vec{x}\ar[drrr]\ar[rrrr]\&\&\&\&
\vec{c'}\ar[rrrr]\&\&\&\&
r_{\vec{c'}}(\vec{x}) \\
\&\&\&\vec{c}\ar[drrr] \ar[ur] \\
\&\&\&\&\&\&r_{\vec{c}}(\vec{x}) \arrow{uurr}[swap]{2({\vec{c'}}-{\vec{c}})} \&
\end{tikzcd}
\]

 \begin{itemize}[<+->]
\item So: $180^\circ$-rotations around different centers differ a translation
of twice the difference between those centers.
\item For every two $(\exists,\stsixe)$-midpoints 
$\vec{c},\vec{c'}$ we do have
the translation $2({\vec{c'}}-{\vec{c}})$ as a symmetry  
\item So we need only one $180^\circ$-rotation (generator $r$)
\end{itemize}
%\vspace*{1cm}
Presentation of the group of symmetries of the carpet:
\[
\langle d,\aa,r \mid d\aa=\aa d, 1=rr=rdrd=r\aa r\aa\rangle
\]
(The latter two relations can be seen by considering $\rhd\!\lhd$)
\end{frame}

\begin{frame}
\frametitle{Univalent group theory}
 \begin{itemize}[<+->]
\item Book in progress:  
\href{https://github.com/UniMath/SymmetryBook}{``{\color{blue}Symmetry}''},
jww Ulrik Buchholtz, Pierre Cagne, Bjørn Ian Dundas, Dan Grayson)
\item Group $G$ is given by a pointed connected 1-type $(BG,\sh_G)$
\item Underlying set of $G$ is $UG \defeq (\sh_G  = \sh_G)$ (`symmetries')
\item Group operation on $UG$ is path composition, unit is $\refl_{\sh_G}$
\item AIM: develop group theory on the level of the pointed 1-types,
called `classifying types'
\end{itemize}
Notation: $T_{(t)} \defeq \sum_{x:T}\Trunc{t=x}$, the
component of $T$ at $t:T$
\end{frame}

\begin{frame}
\frametitle{Examples of univalent groups}
 \begin{itemize}[<+->]
\item $(\Set_{(S)},(S,!))$, the symmetric group of $S:\Set$ (we omit "!")
\item $((\sum_{X:\Set}(X\simeq X))_{(\Zet,\Ss)},(\Zet,\Ss))$, 
the group $\ZZ$ of the integers
\item $((\sum_{X:\Set}(X\simeq X))_{(\mathbf{m},\Ss_m)},(\mathbf{m},\Ss))$,
the cyclic group $C_m$ %, $m>0$
\item $((\sum_{X:\Set}(X\simeq X))_{(\mathbf{4},\pi)},(\mathbf{4},\pi))$ 
with $\pi=(01)(23)$, group?
\end{itemize}
Definitions ($X$ is a non-empty set):
\begin{itemize}[<+->]
\item a \emph{cycle} is a pair $(X,\pi)$ 
with $\pi:X\simeq X$ `connecting everything'
\item a \emph{bicycle} is a pair $(X,\pi,\rho)$ 
with $\pi,\rho:X\simeq X$ `together connecting everything'
\item a \emph{tricycle} is a pair $(X,\pi,\rho,\sigma)$ 
with $\pi,\rho,\sigma:X\simeq X$ `together connecting everything'
\end{itemize}
\end{frame}

\begin{frame}%[fragile]
\frametitle{A different presentation (Ulrik Buchholtz)}

Presentation U of the group of symmetries of the carpet:
\[
\langle a,b,c \mid 1=aa=bb=cc=(abc)^2\rangle
\]
Presentation M was:
\[
\langle d,\aa,r \mid d\aa=\aa d, 1=rr=(rd)^2=(r\aa)^2\rangle
\]
\vspace*{-0.5cm}
\begin{itemize}[<+->]
\item Why are these isomorphic?
\item NB This decision problem is in general undecidable
\item From U to M: $a\mapsto r\aa, ~ b\mapsto rd, ~ c\mapsto r$,
only non-trivial case: $(r\aa rd r)^2 = r\aa r(d \aa) rd r
= r\aa r(\aa d) r d r = drdr=rrdrdr=1$
\item From M to U: $d\mapsto cb, ~ \aa\mapsto ac, ~ r\mapsto c$,
only non-trivial case: $(cb)(ac)=(cba)c=(abc)c=(ac)(cb)$
\item Round trips conjugate $a\mapsto cac$ and $\aa\mapsto r\aa r$,
do them twice
\end{itemize}
\end{frame}
